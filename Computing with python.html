List:-

# Creating a list

my_list = [1, 2, 3, 4, 5]

 

# Accessing elements

print(my_list[0])  # Output: 1

 

# Modifying elements

my_list[0] = 10

 

# Adding elements

my_list.append(6)

 

# Removing elements

my_list.remove(3)

 

# Iterating through a list

for item in my_list:

    print(item)

 

Tuples:-

 

# Creating a tuple

# Accessing elements

Dictionaries:-

 

# Creating a dictionary

my_dict = {'name': 'John', 'age': 25, 'city': 'New York'}

 

# Accessing values

print(my_dict['name'])  # Output: John

 

# Modifying values

my_dict['age'] = 26

 

# Adding new key-value pairs

my_dict['gender'] = 'Male'

 

# Iterating through a dictionary

for key, value in my_dict.items():

    print(key, value)

 

Numpy:-

print(np.sum(my_array))

Pandas :-

 

import pandas as pd

 

# Creating a pandas DataFrame

data = {'Name': ['John', 'Alice', 'Bob'],

        'Age': [25, 30, 22]}

df = pd.DataFrame(data)

 

# Viewing the DataFrame

print(df)

 

# Accessing columns

print(df['Name'])

 

Loc iloc:-

# Using loc to access data by label

print(df.loc[0])

 

# Using iloc to access data by index

print(df.iloc[0])

 

Finding repeated values:-

# Finding repeated values in a list

my_list = [1, 2, 3, 4, 1, 2, 5]

print(set([x for x in my_list if my_list.count(x) > 1]))

 

Finding missing values null values and how to fill them:-

 

# Finding missing values in a DataFrame

print(df.isnull())

 

# Filling missing values with a specific value

df.fillna(0, inplace=True)

 

Converting categorical to non categorical:-

 

# Converting a categorical column to non-categorical

df['Age'] = df['Age'].astype('int')

 

How to read CSV files:-

# Reading a CSV file into a DataFrame

csv_data = pd.read_csv('your_file.csv')

 

How to visualise plt.plot plt.scatter:-
